void solution(){
	
	string s;
	cin >> s;
	stack<int> stc;
	int ans = 0;
	FOR(i, 0, s.length()){
		if(s[i] == '(') stc.push(i);
		else{
			if(stc.empty()){
				stc.push(i);
				continue;
			}
			
			if(s[stc.top()] == '('){
				stc.pop();
				ans = max(ans, stc.empty()? i + 1 : i - stc.top());
			}
			else stc.push(i);
		}
		
	}
	cout << ans << endl;
	
}
